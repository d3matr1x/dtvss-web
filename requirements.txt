# Redis backend for Flask-Limiter on Railway (RAIL-02).
# Railway's private networking is IPv6; redis>=5.0 handles this natively.
# If REDIS_URL isn't set, Flask-Limiter falls back to memory:// (per-worker,
# resets on restart) — safe for local dev, not production.
redis>=5.0.0

# urllib3 — DNS-pinned HTTPS for SSRF guard in security.py.
# The stdlib urllib.request does its own DNS resolution at connection time,
# which left a TOCTOU window between validate_external_url's resolution and
# urlopen's connect. CodeQL alert #20 (py/partial-ssrf) flagged this.
# urllib3.connection.HTTPSConnection takes server_hostname (SNI) and
# assert_hostname (cert validation) as separate kwargs, so we connect to
# the validator-returned IP while TLS still validates against the hostname.
urllib3==2.6.3
